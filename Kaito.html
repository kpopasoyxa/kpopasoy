import time def game_kesehatan(): # Status awal pemain health = 50 energy = 50 hari = 1 print("=== SELAMAT DATANG DI HEALTHY PLATE CHALLENGE ===") print("Misi: Bertahan hidup selama 3 hari dengan menjaga Health & Energy di atas 0.\n") time.sleep(1) while hari <= 3: print(f"\n--- HARI KE-{hari} ---") print(f"Status Anda -> HP: {health} | Energi: {energy}") # Pilihan Makanan print("\nPilih menu makan siang Anda hari ini:") print("1. Pecel Lele & Es Teh Manis") print("2. Sayur Sop, Dada Ayam Bakar & Air Putih") print("3. Mi Instan Double & Gorengan") pilihan = input("Masukkan nomor pilihan (1/2/3): ") # Logika akibat makanan if pilihan == "1": print("\n[!] Rasanya enak! Energi bertambah, tapi kolesterol sedikit naik.") health -= 10 energy += 30 elif pilihan == "2": print("\n[!] Sangat sehat! Tubuh Anda terasa bugar dan penuh nutrisi.") health += 20 energy += 20 elif pilihan == "3": print("\n[!] Praktis dan kenyang, tapi tubuh Anda kekurangan vitamin.") health -= 25 energy += 15 else: print("\n[!] Anda tidak makan apa-apa hari ini!") health -= 20 energy -= 20 # Pembatasan nilai maksimal status health = min(health, 100) energy = min(energy, 100) # Cek kondisi kalah if health <= 0 or energy <= 0: print("\n=============================================") print(f"GAME OVER! Anda tumbang di Hari ke-{hari}.") print(f"Status Akhir -> HP: {health} | Energi: {energy}") print("Pesan: Perhatikan lagi keseimbangan nutrisi makanan Anda!") print("=============================================") return hari += 1 time.sleep(1.5) # Kondisi menang print("\n=============================================") print("SELAMAT! Anda berhasil menyelesaikan tantangan 3 hari!") print(f"Status Akhir Anda -> HP: {health} | Energi: {energy}") print("Anda telah belajar memilih makanan yang baik untuk tubuh.") print("=============================================") # Menjalankan game game_kesehatan()
